            <script>
                // Array para armazenar os comentários
                var comentarios = [];

                // Função para adicionar um comentário
                function adicionarComentario() {
                    var nome = document.getElementById('name').value;
                    var comentario = document.getElementById('comment-text').value;
                    
                    if (nome.trim() === '' || comentario.trim() === '') {
                        alert('Por favor, preencha todos os campos.');
                        return;
                    }
                    
                    var comentarioFormatado = nome + ': ' + comentario;
                    comentarios.push(comentarioFormatado);
                    exibirComentarios();
                    
                    // Limpar campos
                    document.getElementById('name').value = '';
                    document.getElementById('comment-text').value = '';
                }

                // Função para exibir os comentários
                function exibirComentarios() {
                    var commentSection = document.getElementById('comment-section');
                    commentSection.innerHTML = '';

                    for (var i = 0; i < comentarios.length; i++) {
                        var comentarioElement = document.createElement('p');
                        comentarioElement.textContent = comentarios[i];
                        commentSection.appendChild(comentarioElement);
                    }
                }

                // Adicionar um ouvinte de evento para o formulário
                document.getElementById('comment-form').addEventListener('submit', function (e) {
                    e.preventDefault(); // Evita o envio do formulário padrão
                    adicionarComentario();
                });
            </script>
